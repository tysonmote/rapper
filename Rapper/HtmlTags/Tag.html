def for( path, version, style )
  @cache ||= {}
  @cache[style] ||= {}
  
  if version
    path << "?v=#{version}"
  end
  
  unless @cache[style][path] && @cache[style][path]
    @cache[style][path] = ERB.new( templates[style] ).result( binding )
  end
  
  @cache[style][path]
end
