def setup_helpers
  if Rapper::ViewHelpers.const_defined?( :RAPPER )
    Rapper::ViewHelpers.send( :remove_const, :RAPPER )
  end
  Rapper::ViewHelpers.const_set( :RAPPER, self )
  
  # Merb
  begin
    Merb::Controller.send( :include, Rapper::ViewHelpers )
  rescue NameError; end
end
